01 — はじめに

応用編について

02 — 全体像

デモを支える技術

多数の同じ形（ブロック・タイル・敵）を並べるデモでは、1個ずつ描くと描画命令が増えすぎて重くなります。応用編の主役はこれを解決する Thin Instances です。まずは考え方を掴みましょう。

03 — 大量描画

Thin Instances

setThinInstances(mesh, matrices, count) は、1 つのメッシュを それぞれ別の位置・向き・大きさで大量に描く仕組みです。Minecraft のボクセル、Tetris のブロック、Sandblox の立方体など、応用デモの土台になります。

matrices は 1 インスタンスあたり 16 個の float（列優先の 4×4 ワールド行列）を並べた Float32Array。色を変えたいときは setThinInstanceColors(mesh, colors)（1 個あたり RGBA の 4 float）を使います。

03

波打つ 196 個の箱（1 メッシュ）

14×14＝196 個の箱を 1 つの元メッシュから描き、毎フレーム高さを波打たせます。行列を直接書き換えたら flushThinInstances() で変更を通知します。

03-thin-instances.tstypescript
import {
    createEngine, createSceneContext, createArcRotateCamera, attachControl,
    createHemisphericLight, createDirectionalLight, createBox, createStandardMaterial,
    setThinInstances, setThinInstanceColors, flushThinInstances,
    addToScene, onBeforeRender, registerScene, startEngine,
} from "@babylonjs/lite";

async function main() {
    const canvas = document.getElementById("renderCanvas") as HTMLCanvasElement;
    const engine = await createEngine(canvas);
    const scene = createSceneContext(engine);

    const camera = createArcRotateCamera(-Math.PI / 3, Math.PI / 3.2, 24, { x: 0, y: 0, z: 0 });
    scene.camera = camera;
    attachControl(camera, canvas, scene);

    addToScene(scene, createHemisphericLight([0, 1, 0], 0.7));
    addToScene(scene, createDirectionalLight([-1, -2, -1], 1.0));

    // 「1つの元メッシュ」を大量に描くための箱
    const box = createBox(engine);
    box.material = createStandardMaterial();

    // N×N 個。各インスタンスは 16 float（列優先の 4x4 ワールド行列）
    const N = 14;
    const count = N * N;
    const matrices = new Float32Array(count * 16);
    const colors = new Float32Array(count * 4);
    let i = 0;
    for (let x = 0; x < N; x++) {
        for (let z = 0; z < N; z++) {
            const o = i * 16;
            // 単位行列 + 平行移動（列優先なので平行移動は 12,13,14 番目）
            matrices[o + 0] = 1; matrices[o + 5] = 1; matrices[o + 10] = 1; matrices[o + 15] = 1;
            matrices[o + 12] = (x - (N - 1) / 2) * 1.5;
            matrices[o + 14] = (z - (N - 1) / 2) * 1.5;
            // 位置に応じた色（RGBA）
            const c = i * 4;
            colors[c + 0] = x / N; colors[c + 1] = 0.45; colors[c + 2] = z / N; colors[c + 3] = 1;
            i++;
        }
    }
    setThinInstances(box, matrices, count);
    setThinInstanceColors(box, colors);
    addToScene(scene, box);

    // 毎フレーム、各インスタンスの高さ(Y)を波打たせる
    let t = 0;
    onBeforeRender(scene, (deltaMs) => {
        t += deltaMs * 0.002;
        for (let k = 0; k < count; k++) {
            const o = k * 16;
            matrices[o + 13] = Math.sin(t + matrices[o + 12] * 0.5 + matrices[o + 14] * 0.5) * 0.9;
        }
        flushThinInstances(box); // 行列を直接書き換えたので変更を通知
    });

    await registerScene(scene);
    await startEngine(engine);
}
main().catch((err) => console.error(err));
⚡

もっと速く：GPU カリング

数万個規模になったら enableThinInstanceGpuCulling(mesh)（registerScene() の前に呼ぶ）で、画面外インスタンスを GPU 側で間引けます。

04 — 質感

PBR・IBL・発光

Tetris のつやつやしたブロックのような質感は、PBR マテリアルと画像ベースの環境光（IBL）の組み合わせで作ります。loadEnvironment() で環境（映り込み）を読み込み、createPbrMaterial() の metallicFactor / roughnessFactor で金属感・つや、emissiveColor で自発光（光るブロック）を表現します。

⚠️

brdfUrl は必須

loadEnvironment() は BRDF LUT 画像を必ず読み込みます。brdfUrl を省くと「The source image could not be decoded」エラーになります。Playground では brdfUrl: "/brdf-lut.png" が使えます。

04

金属球と発光ブロック

環境（IBL）を読み込み、つやのある金属球と、emissiveColor で光る箱を並べます。

04-pbr-ibl-emissive.tstypescript
import {
    createEngine, createSceneContext, createDefaultCamera, attachControl,
    createHemisphericLight, createSphere, createBox, createPbrMaterial,
    loadEnvironment, addToScene, registerScene, startEngine,
} from "@babylonjs/lite";

async function main() {
    const canvas = document.getElementById("renderCanvas") as HTMLCanvasElement;
    const engine = await createEngine(canvas);
    const scene = createSceneContext(engine);

    addToScene(scene, createHemisphericLight([0, 1, 0], 0.3));

    // 画像ベースの環境光（IBL）。金属やつやの映り込みに必須（Tetris の質感）
    await loadEnvironment(scene, "https://assets.babylonjs.com/core/environments/environmentSpecular.env", {
        brdfUrl: "/brdf-lut.png", // BRDF LUT は必須。Playground がこのパスで配信している
        skyboxUrl: "https://assets.babylonjs.com/core/environments/backgroundSkybox.dds",
        skyboxSize: 1000,
    });

    // つやのある金属球（metallic=1, roughness 低め）
    const metal = createSphere(engine, { segments: 32, diameter: 2 });
    metal.position.set(-1.6, 0, 0);
    metal.material = createPbrMaterial({
        baseColorFactor: [0.9, 0.9, 0.95, 1], metallicFactor: 1, roughnessFactor: 0.15,
    });
    addToScene(scene, metal);

    // 自ら発光する箱（emissiveColor）= 光るブロック
    const glow = createBox(engine);
    glow.position.set(1.6, 0, 0);
    glow.material = createPbrMaterial({
        baseColorFactor: [0.05, 0.05, 0.05, 1], roughnessFactor: 0.4, emissiveColor: [0.1, 0.6, 1.0],
    });
    addToScene(scene, glow);

    // 読み込んだ内容に合わせてカメラを自動配置
    const camera = createDefaultCamera(scene);
    attachControl(camera, canvas, scene);

    await registerScene(scene);
    await startEngine(engine);
}
main().catch((err) => console.error(err));
✨

まぶしさ（Bloom）を足すなら

emissiveColor は「明るい色」を出すだけです。光がにじむ効果は後処理の Bloom（08 節）で足します。

05 — 影

影（ShadowGenerator）

影は「影を落とす側（caster）」と「影を受ける側（receiver）」を指定します。方向性ライトには createEsmDirectionalShadowGenerator() を割り当て、setShadowTaskCasterMeshes() でキャスターを登録、受け手には mesh.receiveShadows = true を設定します。

⚠️

登録関数が変わる

影を使うシーンは registerScene() の代わりに registerSceneWithShadowSupport() で登録します（影のスケジューリングを組み込むため）。

05

地面に落ちる球の影

方向性ライト＋ESM シャドウ。球が caster、地面が receiver です。

05-shadows.tstypescript
import {
    createEngine, createSceneContext, createArcRotateCamera, attachControl,
    createHemisphericLight, createDirectionalLight,
    createSphere, createGround, createStandardMaterial,
    createEsmDirectionalShadowGenerator, setShadowTaskCasterMeshes,
    addToScene, registerSceneWithShadowSupport, startEngine,
} from "@babylonjs/lite";

async function main() {
    const canvas = document.getElementById("renderCanvas") as HTMLCanvasElement;
    const engine = await createEngine(canvas);
    const scene = createSceneContext(engine);

    const camera = createArcRotateCamera(-Math.PI / 2.4, Math.PI / 3, 11, { x: 0, y: 0.5, z: 0 });
    scene.camera = camera;
    attachControl(camera, canvas, scene);

    addToScene(scene, createHemisphericLight([0, 1, 0], 0.4));

    // 影を落とす方向性ライト
    const sun = createDirectionalLight([-1, -2, -1], 1.2);
    addToScene(scene, sun);

    // 影を受ける地面
    const ground = createGround(engine, { width: 14, height: 14 });
    const gm = createStandardMaterial();
    gm.diffuseColor = [0.35, 0.37, 0.4];
    ground.material = gm;
    ground.receiveShadows = true;
    addToScene(scene, ground);

    // 影を落とす球
    const ball = createSphere(engine, { diameter: 2.4 });
    ball.position.set(0, 1.8, 0);
    ball.material = createStandardMaterial();
    addToScene(scene, ball);

    // ESM（指数シャドウマップ）の影生成器をライトに割り当て、キャスターを登録
    sun.shadowGenerator = createEsmDirectionalShadowGenerator(engine, sun, { mapSize: 1024, depthScale: 50 });
    setShadowTaskCasterMeshes(sun.shadowGenerator, [ball]);

    // 影シーンは registerScene ではなく registerSceneWithShadowSupport で登録する
    await registerSceneWithShadowSupport(scene);
    await startEngine(engine);
}
main().catch((err) => console.error(err));
06 — 入力

ピッキング（クリックで選択）

Sandblox のブロック編集のように「クリックした 3D 物体を特定する」には GPU ピッキングを使います。createGpuPicker(scene) で作り、独立関数 pickAsync(picker, x, y)（キャンバス基準のピクセル座標）で pickedMesh や pickedPoint が得られます。picker.pickAsync(...) のようなメソッド呼び出しではない点に注意してください。

06

クリックした箱を赤くする

実行後、箱をクリックすると色が変わります。実行中のマテリアル変更は markMaterialUboDirty() で通知します。

06-picking.tstypescript
import {
    createEngine, createSceneContext, createArcRotateCamera, attachControl,
    createHemisphericLight, createBox, createStandardMaterial, markMaterialUboDirty,
    createGpuPicker, pickAsync, addToScene, registerScene, startEngine,
} from "@babylonjs/lite";

async function main() {
    const canvas = document.getElementById("renderCanvas") as HTMLCanvasElement;
    const engine = await createEngine(canvas);
    const scene = createSceneContext(engine);

    const camera = createArcRotateCamera(-Math.PI / 3, Math.PI / 3, 13, { x: 0, y: 0, z: 0 });
    scene.camera = camera;
    attachControl(camera, canvas, scene);

    addToScene(scene, createHemisphericLight([0, 1, 0], 0.9));

    // 3×3 に別々の箱を並べる
    for (let x = -1; x <= 1; x++) {
        for (let z = -1; z <= 1; z++) {
            const b = createBox(engine);
            b.position.set(x * 2.4, 0, z * 2.4);
            const m = createStandardMaterial();
            m.diffuseColor = [0.4, 0.5, 0.7];
            b.material = m;
            addToScene(scene, b);
        }
    }

    // GPU ピッカー（Sandblox のブロック選択などに使う）
    const picker = createGpuPicker(scene);

    // クリックした箱を赤くする
    canvas.addEventListener("pointerdown", async (ev) => {
        // pickAsync は独立関数。offsetX/offsetY はキャンバス基準の座標
        const info = await pickAsync(picker, ev.offsetX, ev.offsetY);
        if (info.hit && info.pickedMesh && info.pickedMesh.material) {
            info.pickedMesh.material.diffuseColor = [0.95, 0.3, 0.2];
            markMaterialUboDirty(info.pickedMesh.material); // 実行中の色変更は dirty 通知が必要
            console.log("picked point:", info.pickedPoint);
        }
    });

    await registerScene(scene);
    await startEngine(engine);
}
main().catch((err) => console.error(err));
07 — カスタム描画

WGSL シェーダ（ShaderMaterial）

Minecraft の水面、マリオ風の CRT 効果、Freeciv の地形など、独自の見た目は自作シェーダで作ります。Babylon Lite のシェーダは WGSL 専用（GLSL 非対応）。createShaderMaterial() に頂点・フラグメントの WGSL を渡します。

  • system uniform（worldViewProjection など）は WGSL 内で shaderSystem.～
  • 自作 uniform（time など）は shaderUniforms.～、値は setShaderUniform() で更新
  • 頂点入力 VertexInput は attributes から自動生成、エントリ関数は mainVertex / mainFragment
07

参考：頂点を揺らす WGSL シェーダの雛形

API の形を示す参考コードです。mesh.material = wave として使い、毎フレーム time を更新します。

07-shader-material.ts（参考）typescript + wgsl
import { createShaderMaterial, setShaderUniform } from "@babylonjs/lite";

// Babylon Lite のシェーダは WGSL で書く（GLSL は非対応）。
// system uniform（worldViewProjection 等）は shaderSystem.～、
// カスタム uniform（time, color）は shaderUniforms.～ でアクセスする。
const wave = createShaderMaterial({
    name: "wave",
    attributes: ["position", "normal"],
    uniforms: ["worldViewProjection", { name: "time", type: "f32" }, { name: "color", type: "vec3<f32>" }],
    vertexSource: `
        struct VertexOutput {
            @builtin(position) position: vec4<f32>,
            @location(0) vNormal: vec3<f32>,
        };
        @vertex
        fn mainVertex(input: VertexInput) -> VertexOutput {
            var out: VertexOutput;
            var p = input.position;
            p.y = p.y + sin(shaderUniforms.time + p.x * 3.0) * 0.15;
            out.position = shaderSystem.worldViewProjection * vec4<f32>(p, 1.0);
            out.vNormal = input.normal;
            return out;
        }
    `,
    fragmentSource: `
        struct VertexOutput {
            @builtin(position) position: vec4<f32>,
            @location(0) vNormal: vec3<f32>,
        };
        @fragment
        fn mainFragment(input: VertexOutput) -> @location(0) vec4<f32> {
            let light = clamp(dot(normalize(input.vNormal), normalize(vec3<f32>(0.4, 1.0, 0.3))), 0.2, 1.0);
            return vec4<f32>(shaderUniforms.color * light, 1.0);
        }
    `,
});

// mesh.material = wave; addToScene(scene, mesh);
// 毎フレーム：setShaderUniform(wave, "time", t) で時間を渡してアニメーション
08 — 2D と仕上げ

スプライトと後処理

スプライトは常にカメラを向く 2D の板です。DOOM の敵やアイテム、Freeciv のタイル絵に使われます。1 枚のテクスチャ（アトラス）を格子分割し、フレーム番号で絵柄を切り替えます。後処理（Bloom など）は描画後の画面全体にかけるエフェクトで、発光の「にじみ」を作ります。どちらも仕組みが大きいので、ここでは API の入口を示します。

参考：スプライト

sprites.ts（参考）
import { loadSpriteAtlas, createGridSpriteAtlas, createSpriteRenderer } from "@babylonjs/lite";

// スプライト（2Dの板）= DOOM の敵/アイテム、Freeciv のタイル絵に使われる。
// 1枚のテクスチャ（アトラス）を格子状に分割し、フレーム番号で絵柄を選ぶ。
const atlas = await loadSpriteAtlas(engine, "https://example.com/sprites.png", {
    // グリッド分割の指定など（詳細は docs: architecture/32-sprites）
});

const sr = createSpriteRenderer(engine, {
    // レイヤー・ソート順などの設定
});
// レイヤーにスプライトを追加し、位置とフレーム番号を毎フレーム更新する。

参考：Bloom（後処理）

bloom.ts（参考）
import { createBloomPostProcessTask } from "@babylonjs/lite";

// Bloom（後処理）= 明るい部分をにじませて「まぶしさ」を出す。
// Minecraft のグロー、Tetris の発光ブロックの仕上げに使われる。
const bloom = createBloomPostProcessTask(
    { name: "bloom", threshold: 0.8, weight: 0.6, kernel: 64 },
    engine, scene,
);
// 後処理はシーンのフレームグラフに task として組み込む。
// 詳細な配線は docs: architecture/29-post-process を参照。
